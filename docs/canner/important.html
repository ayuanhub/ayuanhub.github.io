完成體驗人數​

SELECT
    ud.custid,
    uc.lineuserid,
    dia.highrisk,
    dia.mbox,
    dia.createdtime
FROM
    dialoglogentity dia
    LEFT JOIN userclue uc ON uc.id = dia.userclueid
    LEFT JOIN (
        -- 因userdata有重複ID，取最新資料
        SELECT
            *
        FROM
            (
                SELECT
                    ROW_Number() OVER (
                        PARTITION BY custid
                        ORDER BY
                            updatetime DESC
                    ) AS sort_ID,
                    *
                FROM
                    userdata
            ) AS temp
        WHERE
            temp.sort_ID = 1
    ) AS ud ON ud.id = uc.userdataid
